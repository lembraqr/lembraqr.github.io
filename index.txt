  <script>
    (function() {
      const params = new URLSearchParams(window.location.search);
      const id = params.get("f");

      if (!id) {
        alert("ID não fornecido na URL.");
        document.getElementById("loading").style.display = "none";
        return;
      }

      fetch("https://decorous-zesty-entree.glitch.me/ver", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ id })
      })
      .then(res => {
        if (!res.ok) throw new Error("Resposta não OK");
        return res.json();
      })
      .then(data => {
        if (data.error) {
          alert(data.error);
          document.getElementById("loading").style.display = "none";
          return;
        }

        // Atualiza o banner e a foto de perfil
        const bannerUrl = (data.banner && data.banner.trim() !== "") ? data.banner : "qr/banner/1.jpg";
        document.getElementById("banner-img").src = bannerUrl;

        if (data.imagens && data.imagens.length > 0) {
          document.getElementById("foto-perfil").src = `data:image/jpeg;base64,${data.imagens[0]}`;
        }

        // Atualiza os dados do painel de perfil (modo Galeria)
        document.getElementById("nome-pessoa").textContent = data.pessoa || "Sem nome";
        document.getElementById("nacido-em").textContent =
          data.biografia && data.biografia["naceu-em"] ? data.biografia["naceu-em"] : "Sem local de nacimento";
        document.getElementById("data-dnf").textContent =
          data.biografia && data.biografia.dnf ? data.biografia.dnf : "Sem DN-DO";

        // Preenche a galeria (ignorando a foto de perfil)
        const galeria = document.getElementById("fotos-galeria");
        galeria.innerHTML = "";
        data.imagens.forEach((base64, index) => {
          if (index === 0) return;
          const item = document.createElement("div");
          item.className = "grid-item";
          const img = document.createElement("img");
          img.src = `data:image/jpeg;base64,${base64}`;
          img.alt = "Foto da Galeria";
          item.appendChild(img);
          galeria.appendChild(item);
        });
        
        // Monta o conteúdo do painel de Biografia
        const biographyContainer = document.getElementById("biography-container");
        biographyContainer.innerHTML = "";
          
        // Cria o painel HEADER com informações adicionais alinhadas à esquerda
        const bioHeader = document.createElement("div");
        bioHeader.className = "biography-header";
          
        const additionalInfo = {
          "Família": data.biografia && data.biografia.familia ? data.biografia.familia : "Silva",
          "Origem": data.biografia && data.biografia.origem ? data.biografia.origem : "Brazileira",
          "Ocupação": data.biografia && data.biografia["area-atuacao"] ? data.biografia["area-atuacao"] : "Pedreiro",
          "Nascido em": data.biografia && data.biografia["naceu-em"] ? data.biografia["naceu-em"] : "campomourão-PR",
          "DNF": data.biografia && data.biografia.dnf ? data.biografia.dnf : "1969 - 2020"
        };

        Object.keys(additionalInfo).forEach(key => {
          const infoItem = document.createElement("div");
          infoItem.className = "info-item";
          const iconSpan = document.createElement("span");
          iconSpan.className = "icon";
          // Ícone placeholder – substituir conforme desejar
          iconSpan.innerHTML = "&#9733;";
          const textSpan = document.createElement("span");
          textSpan.innerHTML = `<strong>${key}:</strong> ${additionalInfo[key]}`;
          infoItem.appendChild(iconSpan);
          infoItem.appendChild(textSpan);
          bioHeader.appendChild(infoItem);
        });

        // Cria o painel com a história biográfica
        const bioHistory = document.createElement("div");
        bioHistory.className = "biography-history";
        bioHistory.innerHTML = data.biografia && data.biografia.historia ? data.biografia.historia : "História não disponível.";

        // Insere os dois painéis no container de Biografia
        biographyContainer.appendChild(bioHeader);
        biographyContainer.appendChild(bioHistory);

        // Oculta o overlay de carregamento
        document.getElementById("loading").style.display = "none";
      })
      .catch(err => {
        alert("Erro ao carregar os dados: " + err.message);
        document.getElementById("loading").style.display = "none";
      });

      // Evento para exibir o painel de Biografia (apenas esconder os painéis de Perfil e Galeria)
      document.getElementById("btn-biografia").addEventListener("click", function() {
        document.getElementById("profile-info").style.display = "none";
        document.getElementById("fotos-galeria").style.display = "none";
        document.getElementById("biography-container").style.display = "block";
      });

      // Evento para exibir a Galeria (modo original)
      document.getElementById("btn-galeria").addEventListener("click", function() {
        document.getElementById("biography-container").style.display = "none";
        document.getElementById("profile-info").style.display = "block";
        document.getElementById("fotos-galeria").style.display = "grid";
      });
    })();
  </script>
